var allSpecies = [];
var detailsList = [];

//get list of all species
$.each(data, function(key, value) {
  var species = value.species;
  
  if(species.length == 0) {
    return true;
  }
  
  $.each(species, function(key, value) {
    allSpecies.push(value);
  });
});

//remove duplicates
var uniqueSpecies = [];
$.each(allSpecies, function(i, el){
    if($.inArray(el, uniqueSpecies) === -1) uniqueSpecies.push(el);
});

uniqueSpecies.sort();

$.each(uniqueSpecies, function(i, name){
    var formattedName = name.replace(/ /g, '_');
  
  $.get("http://taxo.drilobase.org/index.php?title=" + formattedName, function(data) {
  var family = $(data).find('.species-table b')[0].innerText;
  var genius = $(data).find('.species-table b')[1].innerText;
  var geographicalOrigin = $(data).find('.species-table:nth-child(4) table tr:nth-child(1) td:nth-child(2)').text();
  var distributionStatus = $(data).find('.species-table:nth-child(4) table tr:nth-child(2) td:nth-child(2)').text();
  
  var ecologicalCategory = $(data).find('.species-table:nth-child(5) table tr:nth-child(1) td:nth-child(2)').text();
  
  var details = {
    name: formattedName,
    family: family,
    genus: genius,
    geographicalOrigin: geographicalOrigin,
    distributionStatus: distributionStatus,
    ecologicalCategory: ecologicalCategory
  };
  
  detailsList.push(details);
  $(".data").empty();
  $('.data').text(JSON.stringify(detailsList));
});
});
